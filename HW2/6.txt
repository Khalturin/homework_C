Упростить выражение:
					  _____
		X = (B->A) * (A + B) * (A->C)
		X = (!B + A) * (!A * !B) * (!A + C)
		(!B + A)*(!A * !B) = !B * !A * !B + A * !A * !B = !B * !A + 0, т.к A * !A = 0
		X = (!B * !A) * (!A + C)
		X = !B * !A * !A + !B * !A * C
		X = !B * !A + !B * !A * C, общий множитель !B * !A
		X = !B * !A  * (1 + C)
		1 + C = 1
		X = !B * !A
